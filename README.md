#### 7.实现本地收藏数据的列表显示功能； 等
```
//js中获取所有收藏的数据，并通过页面展示出来，此处是获取本地收藏数据
  requesLocalData:function(){
    var list = [];
    try {
      var value = wx.getStorageSync('collects');
      if (value) {
        // Do something with return value
        var itemArr = JSON.parse(value);
        if (itemArr) {
          list = itemArr;
        }
      }
    } catch (e) {
      // Do something when catch error
      wx.showToast({
        title: '获取缓存数据出错',
        icon: 'fail'
      })
    }
    this.setData({
      collects: list,
    })
  }
```
